component RefuelGasDrop
{
    Description = "Refuels Gas (Place this component on a map object with no colliders)";
    Unlimited = true;
    Uses = 1;
    RefillGas = true;
    RefillAmmo = false;
    RefillHealth = false;
    PlaySound = true;
    Animated = true;

    _cd = 3.0;
    _cdTimer = 0.0;
    _usesLeft = 1;


    function Init()
    {
        self.MapObject.AddBoxCollider("Region", "Characters");
        self._usesLeft = self.Uses;
    }

    function OnNetworkMessage(sender, message)
    {
        data = Json.LoadFromString(message);
        uses = data.Get("uses");

        if (uses != null)
        {
            self._usesLeft = uses;

            if (self._usesLeft <= 0)
            {
                self.MapObject.Active = false;
            }
        }
        
    }

    function OnCollisionEnter(other)
    {
        if (other.Type != "Human" || !other.IsMine || !self.MapObject.Active || self._usesLeft <= 0)
        {
            return;
        }

        if (self._cdTimer <= 0.0)
        {
            self._cdTimer = self._cd;
            used = false;

            if (self.RefillGas)
            {
                if (other.CurrentGas < other.MaxGas)
                {
                    other.CurrentGas = other.MaxGas;
                    used = true;
                }
            }

            if (self.RefillAmmo)
            {
                if (other.Weapon == "Blade")
                {
                    if (other.CurrentBladeDurability < other.MaxBladeDurability || other.currentBlade < other.MaxBlade)
                    {
                        other.CurrentBladeDurability = other.MaxBladeDurability;
                        other.currentBlade = other.MaxBlade;
                        used = true;
                    }
                }
                else
                {
                    if (other.CurrentAmmoRound < other.MaxAmmoRound || other.CurrentAmmoLeft < other.MaxAmmoTotal - other.MaxAmmoRound)
                    {
                        other.CurrentAmmoRound = other.MaxAmmoRound;
                        other.CurrentAmmoLeft = other.MaxAmmoTotal - other.MaxAmmoRound;
                        used = true;
                    }
                }
            }

            if (self.RefillHealth)
            {
                if (other.Health < other.MaxHealth)
                {
                    other.Health = other.MaxHealth;
                    used = true;
                }
            }

            if (self.Unlimited || !used)
            {
                return;
            }

            if (self.PlaySound && used)
            {
                other.PlaySound("Checkpoint");
            }

            data = Dict();
            data.Set("uses", self._usesLeft - 1);
            self.NetworkView.SendMessageAll(Json.SaveToString(data));
        }
    }

    function OnTick()
    {
        if (self._cdTimer > 0.0 && self.MapObject.Active)
        {
            self._cdTimer -= Time.TickTime;
        }

        if (self.Animated && Network.IsMasterClient && self.MapObject.Active)
        {
            self.MapObject.Transform.Rotate(Vector3(0, 10, 0));
        }
    }
}
